Función para leer y escribir archivos en C±.

Agosto, 20, 2019 Chunche95 


#include <stdio.h>
typedef FILE *F; 
typedef char *string;

/* Función completa para el programa. */
int LeeYEscribe(string nombreArchivo, string datos, int funcion){
    F archivo_;
    string leo;
    if(funcion == 1){
        /* 1 Escribe */
        try{
            archivo_=fopen(nombreArchivo,"a+");
        }catch(int s){
            archivo_ = fopen(nombreArchivo,"w");
        }
        fprintf(archivo_,"%s",datos);
    }else if(funcion == 2 ){
        /* 2 Lee */
        try{
            archivo_ = fopen(nombreArchivo, "r");
        }catch(int s){
            printf("Error %d",s);
        }

        while(!feof(archivo_)){
            fscanf(archivo_,"%c",&leo);
            printf("%c", leo);
        }
    }
    fclose(archivo_);
}
